#!/bin/bash
# Script to build and run tests with various sanitizers

set -e

# Colors for output
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m' # No Color

echo -e "${GREEN}========================================${NC}"
echo -e "${GREEN}Smith Test Runner with Sanitizers${NC}"
echo -e "${GREEN}========================================${NC}"

# Find existing host config or use default
HOSTCONFIG=""
if [ -n "$1" ]; then
    HOSTCONFIG="-hc $1"
fi

# Test 1: AddressSanitizer
echo -e "\n${YELLOW}[1/3] Building with AddressSanitizer...${NC}"
BUILD_DIR="build-asan-test"
rm -rf "$BUILD_DIR"

if [ -n "$HOSTCONFIG" ]; then
    python3 config-build.py -bp "$BUILD_DIR" -bt Debug $HOSTCONFIG \
        -DCMAKE_CXX_FLAGS="-fsanitize=address -fno-omit-frame-pointer -g" \
        -DCMAKE_C_FLAGS="-fsanitize=address -fno-omit-frame-pointer -g" \
        -DCMAKE_EXE_LINKER_FLAGS="-fsanitize=address"
else
    echo -e "${RED}Warning: No host config specified. Trying direct CMake...${NC}"
    mkdir -p "$BUILD_DIR"
    cd "$BUILD_DIR"
    cmake .. -DCMAKE_BUILD_TYPE=Debug \
        -DCMAKE_CXX_FLAGS="-fsanitize=address -fno-omit-frame-pointer -g" \
        -DCMAKE_C_FLAGS="-fsanitize=address -fno-omit-frame-pointer -g" \
        -DCMAKE_EXE_LINKER_FLAGS="-fsanitize=address"
    cd ..
fi

echo -e "${GREEN}Building test_solid_mechanics_state_advancer with ASan...${NC}"
cmake --build "$BUILD_DIR" --target test_solid_mechanics_state_advancer -j8

echo -e "${GREEN}Running tests with AddressSanitizer...${NC}"
cd "$BUILD_DIR"
# Set ASan options for better output
export ASAN_OPTIONS=detect_leaks=1:check_initialization_order=1:detect_stack_use_after_return=1
./bin/test_solid_mechanics_state_advancer --gtest_filter="SolidMechanicsMeshFixture.*"
cd ..

# Test 2: UndefinedBehaviorSanitizer
echo -e "\n${YELLOW}[2/3] Building with UndefinedBehaviorSanitizer...${NC}"
BUILD_DIR="build-ubsan-test"
rm -rf "$BUILD_DIR"

if [ -n "$HOSTCONFIG" ]; then
    python3 config-build.py -bp "$BUILD_DIR" -bt Debug $HOSTCONFIG \
        -DCMAKE_CXX_FLAGS="-fsanitize=undefined -fno-omit-frame-pointer -g" \
        -DCMAKE_C_FLAGS="-fsanitize=undefined -fno-omit-frame-pointer -g" \
        -DCMAKE_EXE_LINKER_FLAGS="-fsanitize=undefined"
else
    mkdir -p "$BUILD_DIR"
    cd "$BUILD_DIR"
    cmake .. -DCMAKE_BUILD_TYPE=Debug \
        -DCMAKE_CXX_FLAGS="-fsanitize=undefined -fno-omit-frame-pointer -g" \
        -DCMAKE_C_FLAGS="-fsanitize=undefined -fno-omit-frame-pointer -g" \
        -DCMAKE_EXE_LINKER_FLAGS="-fsanitize=undefined"
    cd ..
fi

echo -e "${GREEN}Building test_solid_mechanics_state_advancer with UBSan...${NC}"
cmake --build "$BUILD_DIR" --target test_solid_mechanics_state_advancer -j8

echo -e "${GREEN}Running tests with UndefinedBehaviorSanitizer...${NC}"
cd "$BUILD_DIR"
./bin/test_solid_mechanics_state_advancer --gtest_filter="SolidMechanicsMeshFixture.*"
cd ..

# Test 3: Valgrind (if available)
echo -e "\n${YELLOW}[3/3] Running with Valgrind (if available)...${NC}"
if command -v valgrind &> /dev/null; then
    # Use existing debug build or create one
    BUILD_DIR="build-debug-test"
    if [ ! -d "$BUILD_DIR" ]; then
        if [ -n "$HOSTCONFIG" ]; then
            python3 config-build.py -bp "$BUILD_DIR" -bt Debug $HOSTCONFIG
        else
            mkdir -p "$BUILD_DIR"
            cd "$BUILD_DIR"
            cmake .. -DCMAKE_BUILD_TYPE=Debug
            cd ..
        fi
        cmake --build "$BUILD_DIR" --target test_solid_mechanics_state_advancer -j8
    fi

    echo -e "${GREEN}Running tests with Valgrind...${NC}"
    cd "$BUILD_DIR"
    valgrind --leak-check=full \
             --show-leak-kinds=all \
             --track-origins=yes \
             --verbose \
             --log-file=valgrind-out.txt \
             ./bin/test_solid_mechanics_state_advancer --gtest_filter="SolidMechanicsMeshFixture.TransientConstantGravity"

    echo -e "${GREEN}Valgrind results saved to ${BUILD_DIR}/valgrind-out.txt${NC}"
    cd ..
else
    echo -e "${YELLOW}Valgrind not found, skipping...${NC}"
fi

echo -e "\n${GREEN}========================================${NC}"
echo -e "${GREEN}All sanitizer tests completed!${NC}"
echo -e "${GREEN}========================================${NC}"
