#!/bin/bash
# Simple test runner without host-config requirement

set -e

echo "========================================="
echo "Simple Smith Test Runner"
echo "========================================="

# Check if we have a working build directory
BUILD_DIR=""
for dir in build build-* ; do
    if [ -d "$dir" ] && [ -f "$dir/CMakeCache.txt" ]; then
        BUILD_DIR="$dir"
        break
    fi
done

if [ -z "$BUILD_DIR" ]; then
    echo "❌ No configured build directory found."
    echo "Please configure a build directory first:"
    echo "  mkdir build && cd build"
    echo "  cmake .."
    echo "  cd .."
    exit 1
fi

echo "✓ Using build directory: $BUILD_DIR"

# Build the test
echo ""
echo "Building test_solid_mechanics_state_advancer..."
cd "$BUILD_DIR"

# Try to build the test
if cmake --build . --target test_solid_mechanics_state_advancer -j8 2>&1 | tail -20; then
    echo "✓ Build succeeded"
else
    echo "❌ Build failed"
    exit 1
fi

# Find the test executable
TEST_EXE=$(find . -name "test_solid_mechanics_state_advancer" -type f -executable 2>/dev/null | head -1)

if [ -z "$TEST_EXE" ]; then
    echo "❌ Could not find test executable"
    exit 1
fi

echo "✓ Found test: $TEST_EXE"

# Run the tests
echo ""
echo "========================================="
echo "Running Tests"
echo "========================================="

# Run all tests
echo ""
echo "Running all tests..."
if "$TEST_EXE" --gtest_color=yes; then
    echo ""
    echo "========================================="
    echo "✅ All tests PASSED!"
    echo "========================================="
else
    echo ""
    echo "========================================="
    echo "❌ Some tests FAILED"
    echo "========================================="
    exit 1
fi
